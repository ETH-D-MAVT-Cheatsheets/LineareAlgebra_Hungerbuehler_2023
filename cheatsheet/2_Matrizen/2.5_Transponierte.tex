(A \cdot B)^T &= B^T \cdot A^T \nonumber \\
(c \cdot A)^T &= c \cdot A^T \nonumber \\
(A^T)^T &= A \nonumber
\end{align}
\end{minipage}
\begin{minipage}[t]{0.49 \columnwidth}
\begin{align}
(A^T)^{-1} &= (A^{-1})^T \nonumber \\
rang(A^T) &= rang(A) \nonumber \\
det(A^T) &= det(A) \nonumber \\
eig(A^T) &= eig(A) \nonumber
\end{align}
\end{minipage}
